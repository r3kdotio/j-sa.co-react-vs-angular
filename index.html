My solution to my requirement in React.js

class Tabs extends React.Component {
  constructor() {
    super();
    this.handleSubmit = this.handleSubmit.bind(this);
    this.comments = [];
  }
  handleSubmit(event) {
    this.comments.push(this.input.value);
    super.forceUpdate();
  }
  render() {
    return (
      <div>
        <form onSubmit={this.handleSubmit}>
          <input type="text" ref={(input) => this.input = input} />
          <input type="submit" value="Submit" />
        </form>
        {this.comments.map((comment, i) =>
          <div key={i}>{comment}</div>
        )}
      </div>
    );
  }
}

My solution to my requirement in Angular

@Component({
  template: `<p>Add your comment here</p>
    <input type="text" #myinput />
    <button type="button" (click) = 'addComment(myinput.value)'>Add input</button><br>
    <li *ngFor="let comment of comments">
        {{comment}}
    </li>`
})
export class DashboardComponent {
  comments = [];
  addComment(comment:string) {
    this.comments.push(comment);
  }
}
